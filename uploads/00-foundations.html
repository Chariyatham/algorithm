Foundations — เริ่มจากศูนย์จริงๆ

หน้านี้สำหรับคนที่ ยังไม่เคยเขียนโปรแกรม หรือ เพิ่งเริ่มเรียน C++ ก่อนเข้าสู่อัลกอริทึมจริง เราต้องเข้าใจพื้นฐานก่อน — อย่ารีบข้าม

Algorithm คืออะไร?

คือ ขั้นตอนการแก้ปัญหา ที่ชัดเจน ทำตามแล้วได้คำตอบเสมอ

ตัวอย่างชีวิตประจำวัน — ทำไข่เจียว:

  1. ตั้งกระทะ ใส่น้ำมัน
  2. ตอกไข่ใส่ชาม คนให้เข้ากัน
  3. เทไข่ลงกระทะ
  4. รอ 1 นาที ใช้ตะหลิวพลิก
  5. รออีก 30 วิ ตักออก

นี่คือ algorithm! มี input (ไข่ + น้ำมัน), process (ขั้นตอน), output (ไข่เจียว)

ในการเขียนโปรแกรม algorithm คือ ลำดับคำสั่ง ที่บอกคอม:

  • รับข้อมูลอะไรมา (input)
  • ประมวลผลอย่างไร (logic)
  • คืนคำตอบอะไรกลับ (output)

Hello World — โปรแกรมแรก

#include <iostream>       // ดึง library สำหรับ input/output
using namespace std;       // ไม่ต้องเขียน std:: หน้าทุกคำ

int main() {                // จุดเริ่มต้นของโปรแกรม
    cout << "Hello!" << endl;
    return 0;
}

วิธี run: เปิด terminal แล้ว

g++ hello.cpp -o hello   # compile

Loop — วน

For loop — รู้จำนวนรอบล่วงหน้า

for (int i = 0; i < 5; i++) {
    cout << i << " ";    // print: 0 1 2 3 4
}

3 ส่วนของ for: (เริ่มต้น; เงื่อนไขวนต่อ; เพิ่มทีละ)

While loop — วนจนกว่าเงื่อนไขจะเท็จ

    n = n / 2;        // หารครึ่ง
    cout << n << " "; // 50 25 12 6 3 1

Array & Vector — เก็บข้อมูลหลายตัว

Array แบบ C เก่า — ขนาดคงที่

int arr[5] = {3, 1, 4, 1, 5};
cout << arr[0];           // 3 (index เริ่มที่ 0)
cout << arr[4];           // 5
arr[2] = 99;              // เปลี่ยนค่า

// วน:
for (int i = 0; i < 5; i++) cout << arr[i] << " ";
ระวัง! array C ไม่ check ขอบเขต — arr[10] ในโค้ดข้างบนจะอ่าน memory ที่ไม่ใช่ของ array (undefined behavior!) ทำให้โปรแกรมพังโดยไม่บอกสาเหตุ

Vector — ขนาดเปลี่ยนได้ (ควรใช้แทน array เสมอ)

#include <vector>

vector<int> v;            // เริ่มจากว่าง
v.push_back(10);          // เพิ่มท้าย → v = [10]
v.push_back(20);          // v = [10, 20]
v.push_back(30);          // v = [10, 20, 30]

cout << v[0] << endl;     // 10
cout << v.size() << endl; // 3

// วน 2 แบบ:
for (int i = 0; i < v.size(); i++) cout << v[i];
for (int x : v) cout << x;         // range-based for (สวยกว่า)

Function — ฟังก์ชัน

int square(int x) {        // ↑รับ int, ↓คืน int
    return x * x;
}

int main() {
    cout << square(5);     // 25
    int y = square(10);    // y = 100
    return 0;
}

ส่ง vector เข้า function — ใส่ & เพื่อไม่ copy:

int sumAll(vector<int>& v) {     // & = ส่ง reference (เร็ว)
    int s = 0;
    for (int x : v) s += x;
    return s;
}
ทำไมต้องใส่ &? ถ้าไม่ใส่ จะ copy vector ทั้งก้อน — vector ขนาด 10⁶ ก็ copy 10⁶ element ช้ามาก การใส่ & เป็น "ส่ง pointer" ไม่ copy — เร็วและประหยัด memory

คิดแบบ Algorithm — Trace ทีละบรรทัด

เขียน algorithm เก่งต้อง อ่านโค้ดออก ลองทำตามมือ:

โจทย์: หาจำนวนเต็มที่มากที่สุดใน array

int findMax(vector<int>& a) {
    int maxVal = a[0];                 // เริ่มจากตัวแรก
    for (int i = 1; i < a.size(); i++) {
        if (a[i] > maxVal) {
            maxVal = a[i];
        }
    }
    return maxVal;
}

Trace บน a = [3, 7, 2, 9, 5]:

เริ่ม: maxVal = 3 (a[0])
i=1: a[1]=7 > 3? ใช่ → maxVal = 7
i=2: a[2]=2 > 7? ไม่
i=3: a[3]=9 > 7? ใช่ → maxVal = 9
i=4: a[4]=5 > 9? ไม่
return 9 ✓

Pattern พื้นฐาน 3 อย่าง

Pattern 1 — วนเก็บค่ารวม (accumulator)

int sum = 0;
for (int x : v) sum += x;        // หาผลรวม

int product = 1;
for (int x : v) product *= x;    // หาผลคูณ

Pattern 2 — วนเก็บค่าดีที่สุด (best so far)

int maxVal = INT_MIN;            // ค่าเริ่มแย่สุด
for (int x : v) maxVal = max(maxVal, x);

int minVal = INT_MAX;
for (int x : v) minVal = min(minVal, x);

Pattern 3 — วนนับ (counter)

int count = 0;
for (int x : v) if (x > 0) count++;     // นับเลขบวก

คิดเป็น algorithm — กระบวนการ 4 step

1
เข้าใจโจทย์ — input คืออะไร? output คืออะไร? เขียนลงกระดาษ
2
ลองด้วยตัวอย่างเล็ก — ทำด้วยมือก่อน เช่นถ้าโจทย์ "หาผลรวม" ลองกับ [1,2,3] ก่อน
3
เขียนเป็น pseudocode — เขียนขั้นตอนเป็นภาษาไทย/อังกฤษก่อน อย่ารีบเขียน C++
4
แปลงเป็น code — แล้ว trace ตรวจสอบกับตัวอย่าง

Bug ที่มือใหม่เจอบ่อย

  • ลืม ; ท้ายบรรทัด — compiler error: expected ';'
  • เขียน = แทน == — เงื่อนไขไม่ทำงานถูก
  • Off-by-one — for (i=0; i<=n; i++) เขียนผิดเป็น i<=n → เกินขอบ
  • Index out of range — a[size()] เกินขอบ array
  • ลืม #include — ใช้ vector โดยไม่ #include <vector>
  • ลืม return — function บอกว่า return int แต่ไม่มี return

แบบฝึก — ลองทำเองก่อนเปิดเฉลย

Q1 เขียนโปรแกรมรับเลข n แล้ว print 1 ถึง n
เฉลย 
#include <iostream>
using namespace std;
int main() {
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++) cout << i << " ";
    return 0;
}
Q2 รับเลข n แล้ว print เลขคู่ตั้งแต่ 1 ถึง n
เฉลย 
int n; cin >> n;
for (int i = 2; i <= n; i += 2) cout << i << " ";
// หรือ: if (i % 2 == 0) cout << i;
Q3 รับ array แล้วนับว่ามีเลข 0 กี่ตัว
เฉลย 
int n; cin >> n;
vector<int> v(n);
for (int i = 0; i < n; i++) cin >> v[i];
int count = 0;
for (int x : v) if (x == 0) count++;
cout << count;
Q4 รับเลข n แล้ว print n! (factorial)
เฉลย 
int n; cin >> n;
long long f = 1;        // ใช้ long long กัน overflow
for (int i = 2; i <= n; i++) f *= i;
cout << f;
// 5! = 120, 10! = 3,628,800, 20! = 2,432,902,008,176,640,000
Q5 รับ array แล้วบอกว่าเรียงน้อย→มากแล้วหรือไม่
เฉลย 
int n; cin >> n;
vector<int> v(n);
for (int i = 0; i < n; i++) cin >> v[i];

bool sorted = true;
for (int i = 1; i < n; i++) {
    if (v[i] < v[i-1]) { sorted = false; break; }
}
cout << (sorted ? "yes" : "no");
พร้อมแล้ว! ถ้าทำ 5 ข้อนี้ได้คล่อง ก็พร้อมเข้าหน้า Big O แล้ว หัวข้อต่อไปจะเริ่มเข้าสู่อัลกอริทึมจริง ๆ
