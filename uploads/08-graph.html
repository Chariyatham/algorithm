Graph Algorithms — เดินใน graph

Graph = node + edge — โมเดลเกือบทุกอย่างเป็น graph ได้: เพื่อนใน Facebook, ถนน, internet, dependency ของ task ในงาน เราจะเรียนการ เก็บ graph และ ท่อง graph (BFS, DFS) แล้วต่อยอดถึง shortest path

คำศัพท์

  • Node / Vertex (V) — จุดยอด
  • Edge (E) — เส้นเชื่อม 2 node
  • Directed — edge มีทิศทาง (เช่น Twitter follow), Undirected — สองทาง (เช่น Facebook friend)
  • Weighted — edge มี cost (เช่น ระยะทาง)
  • Path — sequence ของ node ที่เชื่อมกันด้วย edge
  • Cycle — path ที่กลับมา node เดิม
  • Connected — มี path เชื่อมทุกคู่ของ node

วิธีเก็บ graph: 2 แบบหลัก

1. Adjacency Matrix

adj[i][j] = 1 ถ้ามี edge i → j

int n = 5;
vector<vector<int>> adj(n, vector<int>(n, 0));
adj[0][1] = adj[1][0] = 1;    // undirected: ใส่ทั้ง 2 ฝั่ง
adj[1][2] = adj[2][1] = 1;
// ...
// ถ้ามีน้ำหนัก: ใส่ค่าน้ำหนักแทน 1
  • ✓ เช็ค edge ระหว่าง 2 node = O(1)
  • ✗ memory O(V²) — สิ้นเปลืองถ้า edge น้อย
  • เหมาะกับ graph หนาแน่น

2. Adjacency List (ใช้บ่อยกว่ามาก)

int n = 5;
vector<vector<int>> adj(n);
adj[0].push_back(1); adj[1].push_back(0);
adj[1].push_back(2); adj[2].push_back(1);
// ...

// ถ้า weighted: pair<int,int> (neighbor, weight)
vector<vector<pair<int,int>>> wadj(n);
wadj[0].push_back({1, 5});
  • ✓ memory O(V + E) — ประหยัดถ้า edge น้อย (sparse graph)
  • ✗ เช็ค edge u-v ต้องวน list = O(degree)
  • เหมาะกับ graph ทั่วไปในชีวิตจริง

BFS — Breadth-First Search (ค้นกว้าง)

Idea: เริ่มจาก source → เยี่ยม "เพื่อนบ้าน" ก่อน → แล้วค่อยเพื่อนของเพื่อน → ใช้ queue

void bfs(vector<vector<int>>& adj, int src) {
    int n = adj.size();
    vector<bool> visited(n, false);
    queue<int> q;
    q.push(src);
    visited[src] = true;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        cout << u << " ";                    // visit u
        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                q.push(v);
            }
        }
    }
}
// O(V + E)

BFS ให้ "shortest path" ในแง่จำนวน edge (สำหรับ unweighted graph)

DFS — Depth-First Search (ค้นลึก)

Idea: เริ่มจาก source → เดินลึกไปจุดสุด → ถอยขึ้นเมื่อตัน → ใช้ recursion (หรือ stack)

void dfs(vector<vector<int>>& adj, int u, vector<bool>& visited) {
    visited[u] = true;
    cout << u << " ";
    for (int v : adj[u]) {
        if (!visited[v]) dfs(adj, v, visited);
    }
}
// เรียก: vector<bool> vis(n, false); dfs(adj, src, vis);

Connected Components (จำนวนกลุ่มใน graph)

int countComponents(vector<vector<int>>& adj) {
    int n = adj.size(), count = 0;
    vector<bool> visited(n, false);
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            dfs(adj, i, visited);    // เยี่ยมทุกคนในกลุ่ม
            count++;
        }
    }
    return count;
}

Cycle Detection (Directed graph)

// ใช้สี: 0=ขาว(ยังไม่เยี่ยม), 1=เทา(กำลังอยู่ในเส้นทาง), 2=ดำ(เสร็จแล้ว)
bool hasCycle(vector<vector<int>>& adj, int u, vector<int>& color) {
    color[u] = 1;                       // เทา
    for (int v : adj[u]) {
        if (color[v] == 1) return true; // เจอเทา → cycle!
        if (color[v] == 0 && hasCycle(adj, v, color)) return true;
    }
    color[u] = 2;                       // ดำ
    return false;
}

Topological Sort (เรียงลำดับ DAG)

ใช้กับ task ที่มี dependency เช่น "ต้องเรียน A ก่อน B"

void topoDFS(vector<vector<int>>& adj, int u, vector<bool>& visited, stack<int>& st) {
    visited[u] = true;
    for (int v : adj[u]) if (!visited[v]) topoDFS(adj, v, visited, st);
    st.push(u);                         // push หลังเยี่ยมเสร็จ
}

vector<int> topoSort(vector<vector<int>>& adj) {
    int n = adj.size();
    vector<bool> visited(n, false);
    stack<int> st;
    for (int i = 0; i < n; i++) if (!visited[i]) topoDFS(adj, i, visited, st);
    vector<int> order;
    while (!st.empty()) { order.push_back(st.top()); st.pop(); }
    return order;
}

Dijkstra — Shortest Path บน Weighted Graph

BFS หา shortest path ได้แค่ unweighted — ถ้า edge มีน้ำหนัก (เช่น ระยะทาง) ใช้ Dijkstra

Idea (greedy): ในแต่ละ step เลือก unvisited node ที่ ใกล้ source ที่สุด → update distance ของเพื่อนบ้าน → mark visited

vector<int> dijkstra(vector<vector<pair<int,int>>>& adj, int src) {
    int n = adj.size();
    vector<int> dist(n, INT_MAX);
    dist[src] = 0;
    // priority_queue ขั้นต่ำ: (distance, node)
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;          // outdated
        for (auto& [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}
// O((V + E) log V)
Dijkstra ใช้ไม่ได้กับ negative edge! ถ้ามี negative edge ใช้ Bellman-Ford O(VE) แทน

MST — Minimum Spanning Tree

ใน connected weighted graph — หา tree ที่เชื่อมทุก node โดยรวมน้ำหนักน้อยสุด

ดูโค้ด Kruskal ในหน้า Greedy — มีอีก algorithm ชื่อ Prim

Prim — เริ่มจาก node 1 ตัว ขยาย tree ทีละ edge

int prim(vector<vector<pair<int,int>>>& adj) {
    int n = adj.size();
    vector<bool> inMST(n, false);
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    pq.push({0, 0});                    // (weight, node)
    int total = 0;
    while (!pq.empty()) {
        auto [w, u] = pq.top(); pq.pop();
        if (inMST[u]) continue;
        inMST[u] = true;
        total += w;
        for (auto& [v, wt] : adj[u])
            if (!inMST[v]) pq.push({wt, v});
    }
    return total;
}
// O(E log V)

Q1 Number of Islands: grid 2D ของ '1' (land) และ '0' (water) — นับจำนวนเกาะ (เกาะ = group ของ 1 ที่เชื่อมกัน 4 ทิศ)
เฉลย — DFS 
void dfs(vector<vector<char>>& g, int r, int c) {
    int m = g.size(), n = g[0].size();
    if (r < 0 || r >= m || c < 0 || c >= n || g[r][c] != '1') return;
    g[r][c] = '0';                      // mark visited (sink the island)
    dfs(g, r+1, c); dfs(g, r-1, c);
    dfs(g, r, c+1); dfs(g, r, c-1);
}

int numIslands(vector<vector<char>>& g) {
    int count = 0;
    for (int r = 0; r < g.size(); r++)
        for (int c = 0; c < g[0].size(); c++)
            if (g[r][c] == '1') { dfs(g, r, c); count++; }
    return count;
}
Q2 Shortest Path in Maze: grid 0/1 — 0 ผ่านได้, 1 กำแพง — หาเส้นทางสั้นสุดจาก (0,0) ไป (m-1, n-1)
เฉลย — BFS 
int shortestPath(vector<vector<int>>& g) {
    int m = g.size(), n = g[0].size();
    if (g[0][0] || g[m-1][n-1]) return -1;
    vector<vector<int>> dist(m, vector<int>(n, -1));
    queue<pair<int,int>> q;
    q.push({0, 0}); dist[0][0] = 0;
    int dr[] = {-1, 1, 0, 0}, dc[] = {0, 0, -1, 1};
    while (!q.empty()) {
        auto [r, c] = q.front(); q.pop();
        if (r == m-1 && c == n-1) return dist[r][c];
        for (int k = 0; k < 4; k++) {
            int nr = r + dr[k], nc = c + dc[k];
            if (nr >= 0 && nr < m && nc >= 0 && nc < n
                && g[nr][nc] == 0 && dist[nr][nc] == -1) {
                dist[nr][nc] = dist[r][c] + 1;
                q.push({nr, nc});
            }
        }
    }
    return -1;
}
Q3 Course Schedule: มี n course มี prerequisite [a, b] = ต้องเรียน b ก่อน a เช็คว่าเรียนครบทุก course ได้หรือไม่ (ไม่มี dependency เป็นวงกลม)
เฉลย — cycle detect ใน directed graph 
bool canFinish(int n, vector<vector<int>>& pre) {
    vector<vector<int>> adj(n);
    for (auto& p : pre) adj[p[1]].push_back(p[0]);
    vector<int> color(n, 0);
    function<bool(int)> dfs = [&](int u) {
        color[u] = 1;
        for (int v : adj[u]) {
            if (color[v] == 1) return true;             // cycle
            if (color[v] == 0 && dfs(v)) return true;
        }
        color[u] = 2;
        return false;
    };
    for (int i = 0; i < n; i++)
        if (color[i] == 0 && dfs(i)) return false;
    return true;
}
Q4 เหตุใด Dijkstra พังเมื่อมี negative edge?
เฉลย

Dijkstra ตั้งสมมติฐานว่า "node ที่ extract จาก PQ ออกมามี dist ที่ final แล้ว"

แต่ถ้ามี negative edge — node ที่ดูใกล้ในตอนนี้ อาจเอื้อมถึงได้สั้นกว่าผ่านเส้นทางอื่นที่ผ่าน negative edge ในอนาคต

Bellman-Ford ทำซ้ำ V-1 รอบเพื่อ guarantee ว่าทุก path ถูกพิจารณา → จัดการ negative ได้
