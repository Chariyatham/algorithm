Sorting Algorithms — เรียงข้อมูล

เรียง array จากน้อยไปมาก ฟังดูง่าย แต่มีอัลกอริทึมเป็นสิบ ๆ แบบ เราจะดู 5 ตัวที่สำคัญที่สุด พร้อมอนิเมชันให้เห็น ทุก compare ทุก swap

ทำไมต้องรู้หลายแบบ?

ในชีวิตจริงเราใช้ std::sort() ซึ่งเป็น quicksort + insertion sort + heapsort ผสมกัน (called introsort) — แต่ถ้าจะเข้าใจว่าทำไมมันเร็ว ต้องรู้พื้นฐาน

นอกจากนี้ เทคนิค ของ sort แต่ละแบบยังเป็น pattern ที่ใช้ในปัญหาอื่น:

  • Merge sort → divide & conquer pattern
  • Quick sort → partition pattern (ใช้ใน selection problem)
  • Insertion sort → online algorithm pattern (sort ขณะรับข้อมูลทีละตัว)

1. Bubble Sort — เรียงแบบฟองอากาศ

เทียบ ทีละคู่ติดกัน ถ้าผิดให้สลับ วน n-1 รอบ ตัวใหญ่จะ "ลอย" ไปท้าย

void bubbleSort(vector<int>& a) {
    int n = a.size();
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (a[j] > a[j+1])
                swap(a[j], a[j+1]);
        }
    }
}
ปรับให้เร็วขึ้น: ใส่ flag swapped ถ้ารอบไหนไม่ swap เลย แสดงว่าเรียงเสร็จแล้ว → break ทำให้ best case (input เรียงอยู่แล้ว) เป็น O(n)

2. Selection Sort — เลือกตัวน้อยที่สุดแล้วเอามาไว้หน้า

void selectionSort(vector<int>& a) {
    int n = a.size();
    for (int i = 0; i < n - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < n; j++)
            if (a[j] < a[minIdx]) minIdx = j;
        swap(a[i], a[minIdx]);
    }
}

ทุกรอบหาตัวที่น้อยสุดในส่วนที่ยังไม่เรียง แล้วเอามาวางตำแหน่ง i → swap แค่ n-1 ครั้ง (ดีกว่า bubble) แต่ compare ก็ยัง O(n²)

3. Insertion Sort — ค่อย ๆ แทรกเข้าที่

void insertionSort(vector<int>& a) {
    int n = a.size();
    for (int i = 1; i < n; i++) {
        int key = a[i];
        int j = i - 1;
        while (j >= 0 && a[j] > key) {
            a[j+1] = a[j];   // เลื่อนไปขวา
            j--;
        }
        a[j+1] = key;        // แทรก
    }
}

เหมือนเรียงไพ่ในมือ — หยิบไพ่ใบใหม่ขึ้นมาแล้วเสียบเข้าตำแหน่งที่ถูกต้อง เร็วมากเมื่อ array เกือบเรียงอยู่แล้ว (best case O(n))

4. Merge Sort — แบ่งครึ่ง แล้วรวม (Divide & Conquer)

Idea: แบ่ง array ครึ่ง → sort ทั้งสองครึ่ง (recursive) → merge เข้าด้วยกัน

// from labSortLab2/merge_sort.cpp
void merge(vector<int>& arr, int l, int m, int r) {
    int n1 = m - l + 1, n2 = r - m;
    vector<int> L(n1), R(n2);
    for (int i = 0; i < n1; i++) L[i] = arr[l + i];
    for (int j = 0; j < n2; j++) R[j] = arr[m + 1 + j];

    int i = 0, j = 0, k = l;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) arr[k++] = L[i++];
        else              arr[k++] = R[j++];
    }
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

void mergeSort(vector<int>& arr, int l, int r) {
    if (l >= r) return;             // base: 1 element
    int m = l + (r - l) / 2;
    mergeSort(arr, l, m);            // sort ครึ่งซ้าย
    mergeSort(arr, m + 1, r);        // sort ครึ่งขวา
    merge(arr, l, m, r);             // รวม
}
ทำไม O(n log n)?
  • ระดับลึก (depth) ของการแบ่ง = log₂(n)
  • ในแต่ละระดับ ต้องวน element ทั้งหมดเพื่อ merge = n
  • รวม = n × log n = O(n log n) ✓

5. Quick Sort — เลือก pivot แล้วแบ่ง

Idea: เลือก element 1 ตัวเป็น pivot → ย้ายตัวที่น้อยกว่าไปซ้าย ใหญ่กว่าไปขวา → sort 2 ฝั่ง recursive

// from labSortLab2/quick_sort.cpp
int partition(vector<int>& arr, int low, int high) {
    int pivot = arr[high];          // ใช้ตัวสุดท้ายเป็น pivot
    int i = low - 1;
    for (int j = low; j <= high - 1; j++) {
        if (arr[j] < pivot) {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[high]);    // ย้าย pivot มาตรงกลาง
    return i + 1;                    // คืน index ของ pivot
}

void quickSort(vector<int>& arr, int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);    // sort ฝั่งซ้าย
        quickSort(arr, pi + 1, high);   // sort ฝั่งขวา
    }
}
Worst case O(n²): ถ้า input เรียงอยู่แล้ว และเลือก pivot ตัวสุดท้ายเสมอ → partition ไม่สมดุล (ฝั่งหนึ่งว่าง อีกฝั่งเต็ม) → recursion depth = n
วิธีแก้: สุ่ม pivot หรือใช้ "median-of-three"

เลือก sort ตัวไหนเมื่อไหร่?

  • n เล็ก (≤ 30): insertion sort เร็วสุด (overhead น้อย)
  • n กลาง-ใหญ่: quick sort (เร็วที่สุดในทางปฏิบัติ, cache friendly)
  • ต้องการ stable: merge sort
  • memory จำกัดมาก: heap sort หรือ in-place quick sort
  • ในชีวิตจริง: ใช้ std::sort() ไปเลย

Q3 ทำไม merge sort ต้องใช้ memory O(n) แต่ quick sort ใช้แค่ O(log n)?
เฉลย
  • Merge sort ตอน merge ต้อง L, R เก็บข้อมูลชั่วคราว — รวมขนาดทั้งหมดในแต่ละ depth = n
  • Quick sort ทำ partition in-place (เปลี่ยนใน array เดิม) — memory ที่ใช้คือ call stack ซึ่งลึกสุด log n ตอน partition แบ่งครึ่งสมบูรณ์
Q4 (จาก labSortLab2/lab2_q4.cpp)

กำหนด array ของจำนวนเต็ม จงหา ความยาวของ subsequence ที่เป็นเลขติดกัน ที่ยาวที่สุด เช่น [100, 4, 200, 1, 3, 2] → 1,2,3,4 → ยาว 4

เฉลย (วิธี sort) 
int longestConsecutive(vector<int>& arr) {
    if (arr.empty()) return 0;
    sort(arr.begin(), arr.end());
    int maxLen = 1, cur = 1;
    for (int i = 1; i < (int)arr.size(); i++) {
        if (arr[i] == arr[i-1] + 1) cur++;
        else if (arr[i] != arr[i-1]) {
            maxLen = max(maxLen, cur);
            cur = 1;
        }
    }
    return max(maxLen, cur);
}
// O(n log n) จาก sort + O(n) วน → O(n log n) รวม
