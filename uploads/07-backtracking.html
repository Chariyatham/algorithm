Backtracking — เดินไป ถ้าตันก็ถอย

Backtracking = ลองทุกความเป็นไปได้แบบ DFS แต่ ตัดต้นไม้ทิ้ง เมื่อรู้ว่าเส้นทางนี้ไม่มีทางได้คำตอบ ลด search space ได้มหาศาล เป็นเทคนิคที่ใช้แก้ N-Queens, Sudoku, permutation, subset

หลักคิด: state-space tree

นึกภาพการเดินใน ต้นไม้ของ choice

  • แต่ละ node = state ปัจจุบัน (เลือกอะไรไปแล้วบ้าง)
  • แต่ละ branch = choice ที่ทำต่อไป
  • เดิน DFS ลงไป — ถึง leaf = solution
  • เจอ เส้นที่ valid ไม่ได้ → backtrack (ถอยขึ้น แล้วลอง branch อื่น)
void backtrack(state) {
    if (is_solution(state)) {
        record(state);
        return;
    }
    for (each choice in candidates(state)) {
        if (is_valid(state, choice)) {       // ← pruning!
            apply(state, choice);
            backtrack(state);
            undo(state, choice);             // ← ถอย
        }
    }
}

ตัวอย่างที่ 1 — Permutation (จาก lab/p_fo_n.cpp)

สร้าง permutation ของ {1, 2, ..., n} ทั้งหมด

void findPermutations(vector<int>& nums, int left, int right) {
    if (left == right) {                       // base: leaf
        for (int i = 0; i <= right; i++) cout << nums[i] << " ";
        cout << endl;
        return;
    }
    for (int i = left; i <= right; i++) {
        swap(nums[left], nums[i]);             // เลือก nums[i] เป็นตำแหน่ง left
        findPermutations(nums, left + 1, right);
        swap(nums[left], nums[i]);             // backtrack — สลับกลับ
    }
}

เห็น tree: n=3 มี 3! = 6 permutation

ตัวอย่างที่ 2 — N-Queens

วาง n ราชินีบนกระดาน n×n โดย ห้ามอยู่ในแถว/คอลัมน์/แนวทแยงเดียวกัน

// pattern จาก lab/w3.cpp: ใช้ permutation แล้วเช็ค diagonal
// X[i] = column ของ queen ที่อยู่ row i (1-indexed)

bool isValid(vector<int>& X, int end) {
    for (int i = 1; i < end; i++)
        for (int j = i + 1; j <= end; j++)
            if (abs(i - j) == abs(X[i] - X[j]))    // diagonal!
                return false;
    return true;
}

void permute(vector<int>& X, int start, int end) {
    if (start == end) {
        if (isValid(X, end)) {
            for (int i = 1; i <= end; i++) cout << X[i] << " ";
            cout << endl;
        }
        return;
    }
    for (int i = start; i <= end; i++) {
        swap(X[start], X[i]);
        permute(X, start + 1, end);
        swap(X[start], X[i]);
    }
}
เทคนิคปรับ: วิธีของ w3.cpp สร้าง permutation ทุกตัวก่อนค่อยเช็ค → O(n!)
ดีกว่า: เช็คตอน กำลังวาง queen แต่ละตัว ถ้าทำ row นี้ valid ไม่ได้ ก็ตัดทิ้งไม่ต้อง recurse 
// version ที่ prune เร็ว
bool safe(vector<int>& pos, int row, int col) {
    for (int i = 0; i < row; i++) {
        if (pos[i] == col) return false;                // column ซ้ำ
        if (abs(pos[i] - col) == abs(i - row)) return false;  // diagonal
    }
    return true;
}

void solve(vector<int>& pos, int row, int n, int& count) {
    if (row == n) { count++; return; }
    for (int col = 0; col < n; col++) {
        if (safe(pos, row, col)) {
            pos[row] = col;
            solve(pos, row + 1, n, count);
            // pos[row] = -1; // ไม่จำเป็น เพราะรอบหน้าจะ overwrite
        }
    }
}

ตัวอย่างที่ 3 — Subset Sum (จาก lab/w2.cpp)

มี array A — หา subset ที่ผลรวมเท่า target

// pattern แบบ "เลือก / ไม่เลือก" element ทีละตัว
void subset(vector<int>& A, vector<int>& chosen, int idx, int sum, int target) {
    if (idx == A.size()) {
        if (sum == target) {
            for (int i = 0; i < chosen.size(); i++)
                if (chosen[i]) cout << A[i] << " ";
            cout << endl;
        }
        return;
    }
    // ทาง 1: เอา A[idx]
    chosen[idx] = 1;
    subset(A, chosen, idx + 1, sum + A[idx], target);
    // ทาง 2: ไม่เอา
    chosen[idx] = 0;
    subset(A, chosen, idx + 1, sum, target);
}
Pruning: ถ้า sum > target ตัดสาขานี้ทิ้งเลย (ถ้า A มีแต่จำนวนบวก) — ทำให้เร็วขึ้นได้มากใน input ใหญ่ 
// pruned version
void subsetPruned(vector<int>& A, vector<int>& chosen, int idx, int sum, int target) {
    if (sum > target) return;          // ← prune!
    if (idx == A.size()) {
        if (sum == target) print(...);
        return;
    }
    chosen[idx] = 1;
    subsetPruned(A, chosen, idx + 1, sum + A[idx], target);
    chosen[idx] = 0;
    subsetPruned(A, chosen, idx + 1, sum, target);
}

ตัวอย่างที่ 4 — 0/1 Knapsack แบบ backtracking (lab/w4.cpp)

เหมือน subset sum แต่ track max value ที่ weight ไม่เกิน capacity

int n, capacity, maxV = -1;
vector<int> w, v, x, best_x;

void solve(int idx) {
    if (idx == n) {
        int curW = 0, curV = 0;
        for (int i = 0; i < n; i++)
            if (x[i]) { curW += w[i]; curV += v[i]; }
        if (curW <= capacity && curV > maxV) {
            maxV = curV; best_x = x;
        }
        return;
    }
    x[idx] = 1; solve(idx + 1);          // เอา
    x[idx] = 0; solve(idx + 1);          // ไม่เอา
}

ปรับให้เร็วขึ้น: bound + branch — คำนวณ max value ที่เป็นไปได้ของกิ่งนี้ ถ้า ≤ maxV ปัจจุบัน ตัดทิ้ง

ตัวอย่างที่ 5 — Seating (lab/seating.cpp)

มีเด็กผู้ชาย m คน เด็กผู้หญิง n คน — เรียงนั่งติดกันให้ g1 ห้ามติด g2

bool isValid(vector<string>& arr) {
    for (int i = 0; i < arr.size() - 1; i++) {
        if ((arr[i] == "g1" && arr[i+1] == "g2") ||
            (arr[i] == "g2" && arr[i+1] == "g1"))
            return false;
    }
    return true;
}

void solve(vector<string>& seats, int left, int right) {
    if (left == right) {
        if (isValid(seats)) print(seats);
        return;
    }
    for (int i = left; i <= right; i++) {
        swap(seats[left], seats[i]);
        solve(seats, left + 1, right);
        swap(seats[left], seats[i]);
    }
}

Q1 Combination: เขียน function print ทุก combination C(n, k) เช่น n=4, k=2 → {1,2}, {1,3}, {1,4}, {2,3}, {2,4}, {3,4}
เฉลย 
void comb(int n, int k, int start, vector<int>& cur) {
    if (cur.size() == k) {
        for (int x : cur) cout << x << " ";
        cout << endl;
        return;
    }
    for (int i = start; i <= n; i++) {
        cur.push_back(i);
        comb(n, k, i + 1, cur);    // i+1 — กัน duplicate
        cur.pop_back();             // backtrack
    }
}
Q2 Word Break: ให้ string s กับ dict — เช็คว่าตัด s เป็นคำใน dict ได้หรือไม่ เช่น s="catsanddog", dict={cat, cats, and, sand, dog} → ใช่
เฉลย 
bool wordBreak(string s, set<string>& dict, int start) {
    if (start == s.size()) return true;
    for (int len = 1; len <= s.size() - start; len++) {
        string sub = s.substr(start, len);
        if (dict.count(sub) && wordBreak(s, dict, start + len))
            return true;
    }
    return false;
}
// O(2ⁿ) worst case — ปรับให้เร็วได้ด้วย memo → DP
Q3 Sudoku Solver: เติม 9×9 sudoku ให้ครบ
เฉลย — เน้น pruning 
bool isValid(vector<vector<char>>& b, int r, int c, char ch) {
    for (int i = 0; i < 9; i++) {
        if (b[r][i] == ch) return false;
        if (b[i][c] == ch) return false;
        int br = 3*(r/3) + i/3, bc = 3*(c/3) + i%3;
        if (b[br][bc] == ch) return false;
    }
    return true;
}

bool solve(vector<vector<char>>& b) {
    for (int r = 0; r < 9; r++) {
        for (int c = 0; c < 9; c++) {
            if (b[r][c] != '.') continue;
            for (char ch = '1'; ch <= '9'; ch++) {
                if (isValid(b, r, c, ch)) {
                    b[r][c] = ch;
                    if (solve(b)) return true;
                    b[r][c] = '.';        // backtrack
                }
            }
            return false;                  // ลองครบ 1-9 ไม่ได้ → ตัน
        }
    }
    return true;                            // เต็มกระดานทั้งหมด
}
