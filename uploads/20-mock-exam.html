Q1วิเคราะห์ Big O15 คะแนน

หา Big O ของแต่ละข้อ พร้อมอธิบาย:

  1. for (int i = 1; i < n; i = i * 3)
    sum++;
  2. for (int i = 0; i < n; i++)
    for (int j = i; j < n; j = j * 2)
        sum++;
  3. int f(int n) {
    if (n <= 1) return 1;
    return 2 * f(n - 1) + n;
}
เฉลย Q1

(a) i = 1, 3, 9, 27, ... → log₃(n) รอบ → O(log n)

(b) i: n รอบ; j เริ่มที่ i, *= 2 → log(n/i) รอบ

รวม = Σᵢ log(n/i) ≤ n · log n → O(n log n)

(c) T(n) = 2T(n−1) + n

Substitution: T(n) = 2[2T(n−2) + (n−1)] + n = 4T(n−2) + 2(n−1) + n

= 2^k · T(n−k) + Σ 2^i · (n−i) → ที่ k=n: T(n) = 2ⁿ + ... → O(2ⁿ)

Q2Recurrence + Master Theorem15 คะแนน

แก้ recurrence ต่อไปนี้ (ใช้ Master Theorem หรือ substitution):

  1. T(n) = 4T(n/2) + n
  2. T(n) = 2T(n/4) + √n
  3. T(n) = 7T(n/2) + n²
เฉลย Q2

(a) a=4, b=2, d=1 → log₂(4) = 2 > 1 → กรณี 1 → O(n²)

(b) a=2, b=4, d=½ → log₄(2) = ½ = d → กรณี 2 → O(√n · log n)

(c) a=7, b=2, d=2 → log₂(7) ≈ 2.81 > 2 → กรณี 1 → O(n^2.81) (Strassen!)

ใช้ ตัวสุดท้าย เป็น pivot — แสดงทุก partition + recursive call

เฉลย Q3 
quickSort([5,2,9,1,7,3], 0, 5)
  partition: pivot = 3
    i=-1, j=0..4
    j=0: 5<3? no
    j=1: 2<3? yes → i=0, swap arr[0]↔arr[1] → [2,5,9,1,7,3]
    j=2: 9<3? no
    j=3: 1<3? yes → i=1, swap arr[1]↔arr[3] → [2,1,9,5,7,3]
    j=4: 7<3? no
    swap arr[2]↔arr[5] (pivot) → [2,1,3,5,7,9]
    return pi = 2

  quickSort([2,1,3,5,7,9], 0, 1)
    partition: pivot = 1
      j=0: 2<1? no
      swap arr[0]↔arr[1] → [1,2,3,5,7,9]
      return pi = 0
    quickSort([..], 0, -1) → base
    quickSort([..], 1, 1) → base

  quickSort([..], 3, 5)
    partition: pivot = 9
      j=3: 5<9 → i=3
      j=4: 7<9 → i=4
      no swap needed
      return pi = 5
    quickSort([..], 3, 4)
      partition: pivot = 7
        j=3: 5<7 → i=3
        no swap
        return pi = 4
      quickSort([..], 3, 3) → base
      quickSort([..], 5, 4) → base
    quickSort([..], 6, 5) → base

Q4Karatsuba Multiplication20 คะแนน

คำนวณ X × Y โดยที่ X = 1234, Y = 5678 ใช้ Karatsuba — แสดงทุก step + recursion

เฉลย Q4

Level 1: X = 1234, Y = 5678, n = 4, m = 2, 10² = 100

  • a = 12, b = 34
  • c = 56, d = 78
  • z2 = 12 · 56
  • z0 = 34 · 78
  • z1 = (12+34) · (56+78) = 46 · 134

Level 2: 12 · 56 (a=1, b=2, c=5, d=6)

  • z2 = 1·5 = 5, z0 = 2·6 = 12, z1 = 3·11 = 33
  • middle = 33 − 5 − 12 = 16
  • 12·56 = 5·100 + 16·10 + 12 = 672

34 · 78 (a=3, b=4, c=7, d=8)

  • z2 = 21, z0 = 32, z1 = 7·15 = 105
  • middle = 105 − 21 − 32 = 52
  • 34·78 = 21·100 + 52·10 + 32 = 2652

46 · 134 (a=4, b=6, c=13, d=4) — special: c = 13 (2 หลัก)

  • คำนวณคร่าว ๆ: 46·134 = 6164

รวม Level 1:

  • middle = z1 − z2 − z0 = 6164 − 672 − 2652 = 2840
  • X·Y = 672·10000 + 2840·100 + 2652
  • = 6720000 + 284000 + 2652 = 7,006,652

ตรวจ: 1234 × 5678 = 7,006,652 ✓

Q5เขียน Merge Sort30 คะแนน

เขียนฟังก์ชัน mergeSort และ merge เป็นภาษา C++ ครบ — รวม main

+ วิเคราะห์ T(n) ในรูปของ recurrence + แก้

เฉลย Q5 
#include <iostream>
#include <vector>
using namespace std;

void merge(vector<int>& arr, int l, int m, int r) {
    int n1 = m - l + 1, n2 = r - m;
    vector<int> L(n1), R(n2);
    for (int i = 0; i < n1; i++) L[i] = arr[l + i];
    for (int j = 0; j < n2; j++) R[j] = arr[m + 1 + j];

    int i = 0, j = 0, k = l;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) arr[k++] = L[i++];
        else              arr[k++] = R[j++];
    }
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

void mergeSort(vector<int>& arr, int l, int r) {
    if (l >= r) return;
    int m = l + (r - l) / 2;
    mergeSort(arr, l, m);
    mergeSort(arr, m + 1, r);
    merge(arr, l, m, r);
}

int main() {
    vector<int> arr = {16, 25, 2, 54, 36, 9, 12, 66};
    mergeSort(arr, 0, arr.size() - 1);
    for (int x : arr) cout << x << " ";
    return 0;
}

Recurrence: T(n) = 2T(n/2) + O(n)

(2 recursive calls each n/2, merge = O(n))

Master Theorem: a=2, b=2, d=1 → log₂(2) = 1 = d → กรณี 2

T(n) = O(n log n)

กลยุทธ์ทำข้อสอบ midterm:
  • ทำข้อ "วิเคราะห์ Big O" ก่อน — ได้คะแนนเร็ว
  • เขียน วิธีคิด ทุกข้อ แม้ไม่แน่ใจ — ได้คะแนนบางส่วน
  • ข้อ trace (sort, Karatsuba) — แสดงทุก step ห้ามข้าม
  • ข้อเขียนโค้ด — เริ่มจาก main + signature + base case ก่อนเสมอ

Q5Coding Problem (Application)10 คะแนน

ปัญหา: มีกลุ่มคน n คน นั่งเป็นวงกลม — เลือก k คน ห้ามอยู่ติดกัน ให้ผลรวม value สูงสุด

เขียน code (DP หรือ recursion + memo) — input: vector<int> values, int k

เฉลย Q5 Final 
// สำหรับวงกลม: ลองทั้ง 2 case (เอา/ไม่เอาคนแรก)
int rob(vector<int>& v, int l, int r, int k) {
    int n = r - l + 1;
    if (k == 0) return 0;
    if (n < 2*k - 1) return -1;   // ไม่พอ
    // dp[i][j] = max value เลือก j คนจาก v[l..l+i-1]
    vector<vector<int>> dp(n+1, vector<int>(k+1, 0));
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= min(k, (i+1)/2); j++) {
            dp[i][j] = dp[i-1][j];      // ไม่เอา i
            if (i >= 2)
                dp[i][j] = max(dp[i][j], dp[i-2][j-1] + v[l+i-1]);   // เอา i
            else
                dp[i][j] = max(dp[i][j], v[l+i-1]);
        }
    }
    return dp[n][k];
}

int circularRob(vector<int>& v, int k) {
    int n = v.size();
    // case 1: ไม่เอาคนแรก → เลือกจาก v[1..n-1]
    int c1 = rob(v, 1, n-1, k);
    // case 2: เอาคนแรก → เลือก k-1 จาก v[2..n-2] (กันคนติด)
    int c2 = (k > 0 && n >= 2*k - 1)
           ? v[0] + rob(v, 2, n-2, k-1)
           : 0;
    return max(c1, c2);
}
