Dynamic Programming — จำคำตอบเก่าไว้ใช้

DP คือ "recursion ที่ฉลาด" — แทนที่จะคำนวณซ้ำเดิม ๆ เราจำผลลัพธ์ไว้ในตาราง เปลี่ยน 2ⁿ ให้กลายเป็น n² ได้ในหลาย ๆ ปัญหา

เริ่มจาก Fibonacci ที่เราเห็นในบทก่อน

จำได้ว่า fib แบบ recursive ดิบ ๆ คำนวณซ้ำเยอะมาก:

int fib(int n) {
    if (n <= 1) return n;
    return fib(n-1) + fib(n-2);  // O(2ⁿ)
}

fib(5) คำนวณ fib(2) ถึง 3 ครั้ง! fib(50) จะใช้เวลานาน — ทำยังไง?

วิธีที่ 1 — Memoization (top-down)

vector<int> memo(100, -1);

int fibMemo(int n) {
    if (n <= 1) return n;
    if (memo[n] != -1) return memo[n];          // จำได้ → ใช้เลย
    return memo[n] = fibMemo(n-1) + fibMemo(n-2);
}
// ตอนนี้ O(n) — แต่ละ n คำนวณแค่ 1 ครั้ง

วิธีที่ 2 — Tabulation (bottom-up)

int fibDP(int n) {
    if (n <= 1) return n;
    vector<int> dp(n + 1);
    dp[0] = 0; dp[1] = 1;
    for (int i = 2; i <= n; i++)
        dp[i] = dp[i-1] + dp[i-2];
    return dp[n];
}
// O(n) เวลา, O(n) memory (ลดเป็น O(1) ได้ถ้าเก็บแค่ 2 ค่าก่อนหน้า)
รอเริ่ม

หลักคิด — เมื่อไหร่ใช้ DP?

DP ใช้ได้เมื่อปัญหามี 2 คุณสมบัตินี้:

  1. Optimal substructure — คำตอบของปัญหาประกอบจากคำตอบของ sub-problem
    เช่น fib(n) = fib(n-1) + fib(n-2)
  2. Overlapping sub-problems — sub-problem เดียวกันถูกถามซ้ำหลายครั้ง
    เช่น fib(n-2) ถูกเรียกจากทั้ง fib(n) และ fib(n-1)

ถ้าไม่ overlap → ใช้ D&C พอ

วิธีคิด DP step-by-step

  1. นิยาม state — dp[i] หรือ dp[i][j] เก็บอะไร?
  2. หา recurrence — dp[i] = ฟังก์ชันของ state ที่เล็กกว่า
  3. กำหนด base case — state เล็กสุดมีค่าเท่าไหร่?
  4. เลือกลำดับการ fill — ตัวเล็กก่อน หรือ recurse + memo
  5. คืนคำตอบ — อยู่ที่ dp[n]? หรือต้อง max(dp[...])?

ตัวอย่าง 1 — Coin Change (จำนวนเหรียญน้อยสุด)

มีเหรียญ {1, 5, 10, 25} — จำนวนเงิน amount ใช้เหรียญน้อยสุดเท่าไหร่?

state: dp[x] = จำนวนเหรียญน้อยสุดที่ทอนเงิน x ได้

recurrence: dp[x] = 1 + min(dp[x - c]) สำหรับทุกเหรียญ c ≤ x

base: dp[0] = 0

int coinChange(vector<int>& coins, int amount) {
    vector<int> dp(amount + 1, INT_MAX);
    dp[0] = 0;
    for (int x = 1; x <= amount; x++) {
        for (int c : coins) {
            if (c <= x && dp[x - c] != INT_MAX)
                dp[x] = min(dp[x], dp[x - c] + 1);
        }
    }
    return (dp[amount] == INT_MAX) ? -1 : dp[amount];
}
// O(amount × |coins|)

ตัวอย่าง 2 — 0/1 Knapsack

กระเป๋าจุน้ำหนัก W มีของ n ชิ้น แต่ละชิ้นมี weight, value — เลือกใส่ของ (แต่ละชิ้น เอาทั้งชิ้นหรือไม่เอาเลย) ให้ value รวมสูงสุด

state: dp[i][w] = value สูงสุดเมื่อพิจารณาของ i ชิ้นแรก ใช้น้ำหนักไม่เกิน w

recurrence: 2 ทางเลือกสำหรับของชิ้น i:

  • ไม่เอา: dp[i][w] = dp[i-1][w]
  • เอา (ถ้า weight[i] ≤ w): dp[i-1][w - weight[i]] + value[i]

เลือก max ของสองทาง

int knapsack(vector<int>& weight, vector<int>& value, int W) {
    int n = weight.size();
    vector<vector<int>> dp(n + 1, vector<int>(W + 1, 0));
    for (int i = 1; i <= n; i++) {
        for (int w = 0; w <= W; w++) {
            dp[i][w] = dp[i-1][w];                                  // ไม่เอา i
            if (weight[i-1] <= w)
                dp[i][w] = max(dp[i][w],
                              dp[i-1][w - weight[i-1]] + value[i-1]); // เอา i
        }
    }
    return dp[n][W];
}
// O(n × W)

ตัวอย่าง 3 — Longest Common Subsequence (LCS)

"ABCBDAB" และ "BDCABA" → LCS = "BCBA" (ยาว 4)

state: dp[i][j] = LCS ของ s1[0..i-1] กับ s2[0..j-1]

recurrence:

  • ถ้า s1[i-1] == s2[j-1]: dp[i][j] = dp[i-1][j-1] + 1
  • ถ้าไม่: dp[i][j] = max(dp[i-1][j], dp[i][j-1])
int lcs(string& a, string& b) {
    int m = a.size(), n = b.size();
    vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (a[i-1] == b[j-1]) dp[i][j] = dp[i-1][j-1] + 1;
            else dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
        }
    }
    return dp[m][n];
}
// O(m × n)

เทคนิคพิเศษ: ลด memory จาก O(n) → O(1)

หลายปัญหาใช้แค่ 1-2 row ก่อนหน้า ไม่ต้องเก็บทั้ง table

int fibO1(int n) {
    if (n <= 1) return n;
    int prev = 0, cur = 1;
    for (int i = 2; i <= n; i++) {
        int next = prev + cur;
        prev = cur;
        cur = next;
    }
    return cur;
}
// O(n) เวลา, O(1) memory

เทียบ Top-down vs Bottom-up

Top-down (Memo)Bottom-up (Tabulation)
เริ่มจาก n → ลงไปหา 0 เริ่มจาก 0 → ขึ้นไปหา n
recursive + cache iterative + array
คิดง่ายกว่า (เหมือน recursion ปกติ + memo) เร็วกว่านิดหน่อย (ไม่มี call overhead)
ระวัง stack overflow ถ้า n ใหญ่ ไม่มีปัญหา stack
คำนวณเฉพาะ state ที่ต้องการ คำนวณทุก state (อาจเกินจำเป็น)

Q1 Climbing Stairs: ขึ้นบันได n ขั้น แต่ละครั้งก้าวได้ 1 หรือ 2 ขั้น มีกี่วิธี?
เฉลย

เหมือน fibonacci! dp[n] = dp[n-1] + dp[n-2]

int climb(int n) {
    if (n <= 2) return n;
    int a = 1, b = 2;
    for (int i = 3; i <= n; i++) { int c = a + b; a = b; b = c; }
    return b;
}
Q2 House Robber: มีบ้านเรียงเป็นแถว [2, 7, 9, 3, 1] ห้ามขโมย 2 บ้านติดกัน หาเงินสูงสุด (เฉลย: 12 = 2+9+1)
เฉลย

state: dp[i] = เงินสูงสุดเมื่อพิจารณาบ้าน 0..i

recurrence: ขโมยบ้าน i หรือไม่

  • ขโมย: dp[i-2] + nums[i]
  • ไม่ขโมย: dp[i-1]
int rob(vector<int>& nums) {
    int n = nums.size();
    if (n == 0) return 0;
    if (n == 1) return nums[0];
    vector<int> dp(n);
    dp[0] = nums[0];
    dp[1] = max(nums[0], nums[1]);
    for (int i = 2; i < n; i++)
        dp[i] = max(dp[i-1], dp[i-2] + nums[i]);
    return dp[n-1];
}
Q3 Edit Distance: ให้ string a, b — แปลง a → b ด้วย insert/delete/replace ใช้กี่ operation น้อยสุด?
เฉลย

dp[i][j] = ระยะแปลง a[0..i-1] → b[0..j-1]

int editDist(string a, string b) {
    int m = a.size(), n = b.size();
    vector<vector<int>> dp(m+1, vector<int>(n+1));
    for (int i = 0; i <= m; i++) dp[i][0] = i;   // ลบ i ตัว
    for (int j = 0; j <= n; j++) dp[0][j] = j;   // เติม j ตัว
    for (int i = 1; i <= m; i++) {
        for (int j = 1; j <= n; j++) {
            if (a[i-1] == b[j-1]) dp[i][j] = dp[i-1][j-1];
            else dp[i][j] = 1 + min({dp[i-1][j],     // delete
                                     dp[i][j-1],     // insert
                                     dp[i-1][j-1]}); // replace
        }
    }
    return dp[m][n];
}
Q4 เปรียบเทียบ DP กับโค้ดใน lab/w2.cpp (subset sum) ใครเร็วกว่า?
เฉลย

w2.cpp ใช้ backtracking ลองทุก subset → O(2ⁿ)

DP version: dp[i][s] = "ใช้ของ i ตัวแรก ทำ sum = s ได้ไหม?"

bool subsetSum(vector<int>& A, int target) {
    int n = A.size();
    vector<vector<bool>> dp(n+1, vector<bool>(target+1, false));
    for (int i = 0; i <= n; i++) dp[i][0] = true;  // sum=0 เป็นไปได้เสมอ
    for (int i = 1; i <= n; i++) {
        for (int s = 1; s <= target; s++) {
            dp[i][s] = dp[i-1][s];                      // ไม่เอา A[i-1]
            if (A[i-1] <= s)
                dp[i][s] = dp[i][s] || dp[i-1][s - A[i-1]]; // เอา
        }
    }
    return dp[n][target];
}
// O(n × target) — เร็วกว่า 2ⁿ มากเมื่อ target ไม่โต

แต่ถ้าโจทย์ต้อง print ทุก subset backtracking สะดวกกว่า — DP บอกแค่ "ทำได้ไหม"
